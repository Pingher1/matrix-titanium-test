from fastapi import FastAPI, HTTPException
import uuid
from api.schemas import JobRequest, JobResponse
from worker.tasks import run_scan_task

app = FastAPI()
JOB_STORE = {} 

@app.post("/jobs", response_model=JobResponse)
def submit_job(req: JobRequest):
    job_id = str(uuid.uuid4())
    payload = req.dict()
    task = run_scan_task.delay(job_id, payload)
    JOB_STORE[job_id] = {"status": "queued", "task_id": task.id}
    return JobResponse(job_id=job_id, status="queued", dashboard_link=f"/jobs/{job_id}")

@app.get("/jobs/{job_id}", response_model=JobResponse)
def get_job(job_id: str):
    rec = JOB_STORE.get(job_id)
    if not rec: raise HTTPException(status_code=404, detail="Not found")
    return JobResponse(job_id=job_id, status=rec["status"])

INTERVENTIONS = {} 
ACTIVE_SOCKETS: List[WebSocket] = []

class InterventionAlert(BaseModel):
    job_id: str
    screenshot_url: str  # S3 Link to the Captcha
    reason: str = "CAPTCHA_DETECTED"

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept()
    ACTIVE_SOCKETS.append(websocket)
    try:
        while True:
            await websocket.receive_text() # Keep alive
    except:
        ACTIVE_SOCKETS.remove(websocket)

@app.post("/api/intervention/trigger")
async def trigger_intervention(alert: InterventionAlert):
    """ CALLED BY WORKER: 'Help! I am stuck!' """
    alert_id = str(uuid.uuid4())
    INTERVENTIONS[alert_id] = {
        "id": alert_id,
        "job_id": alert.job_id,
        "screenshot": alert.screenshot_url,
        "status": "PENDING_HUMAN",
        "reason": alert.reason
    }
    
    # Broadcast to Cockpit UI
    for ws in ACTIVE_SOCKETS:
        await ws.send_json({"type": "ALERT", "data": INTERVENTIONS[alert_id]})
    
    return {"alert_id": alert_id, "status": "WAITING"}

@app.get("/api/intervention/{alert_id}/status")
async def check_status(alert_id: str):
    """ CALLED BY WORKER: Polling to see if human fixed it """
    if alert_id not in INTERVENTIONS:
        raise HTTPException(404, "Alert not found")
    return {"status": INTERVENTIONS[alert_id]["status"]}

@app.post("/api/intervention/{alert_id}/resolve")
async def resolve_intervention(alert_id: str):
    """ CALLED BY HUMAN (UI): 'I fixed it. Resume.' """
    if alert_id in INTERVENTIONS:
        INTERVENTIONS[alert_id]["status"] = "SOLVED"
        return {"result": "Bot Resumed"}
    raise HTTPException(404, "Alert not found")

import React, { useState, useEffect } from 'react';

export default function MissionControl() {
  const [alerts, setAlerts] = useState([]);

  useEffect(() => {
    // Connect to the Red Phone
    const ws = new WebSocket("ws://localhost:8000/ws");
    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);
      if (msg.type === "ALERT") {
        setAlerts(prev => [...prev, msg.data]);
      }
    };
    return () => ws.close();
  }, []);

  const handleResolve = async (alertId) => {
    // Tell the API we fixed it
    await fetch(`http://localhost:8000/api/intervention/${alertId}/resolve`, { method: "POST" });
    setAlerts(prev => prev.filter(a => a.id !== alertId));
  };

  return (
    <div className="p-10 bg-gray-900 min-h-screen text-white font-mono">
      <h1 className="text-3xl text-red-500 mb-8">🚨 INTERVENTION REQUIRED</h1>
      
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
        {alerts.map(alert => (
          <div key={alert.id} className="border border-red-500 p-4 rounded bg-black">
            <h2 className="text-xl mb-2">JOB ID: {alert.job_id}</h2>
            <p className="text-red-400 mb-4">REASON: {alert.reason}</p>
            
            {/* The Evidence */}
            <img src={alert.screenshot} alt="Captcha" className="mb-4 w-full border border-gray-700"/>
            
            <div className="flex gap-4">
              <button 
                onClick={() => window.open(alert.screenshot, '_blank')}
                className="bg-gray-700 px-4 py-2 rounded hover:bg-gray-600">
                INSPECT
              </button>
              <button 
                onClick={() => handleResolve(alert.id)}
                className="bg-green-600 px-4 py-2 rounded font-bold hover:bg-green-500 w-full">
                ✅ MARK SOLVED & RESUME
              </button>
            </div>
          </div>
        ))}
        {alerts.length === 0 && <p className="text-gray-500">System Nominal. No active threats.</p>}
      </div>
    </div>
  );
}

from typing import List, Dict
from fastapi import WebSocket
from collections import defaultdict

class ConnectionManager:
    def __init__(self):
        self.active: Dict[str, List[WebSocket]] = defaultdict(list)

    async def connect(self, websocket: WebSocket, room: str = "interventions"):
        await websocket.accept()
        self.active[room].append(websocket)

    def disconnect(self, websocket: WebSocket, room: str = "interventions"):
        if websocket in self.active.get(room, []):
            self.active[room].remove(websocket)

    async def broadcast(self, message: dict, room: str = "interventions"):
        for connection in list(self.active.get(room, [])):
            try:
                await connection.send_json(message)
            except:
                self.disconnect(connection, room)

manager = ConnectionManager()

class ConnectionManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []

    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connections.append(websocket)

    def disconnect(self, websocket: WebSocket):
        self.active_connections.remove(websocket)

    async def broadcast(self, message: dict):
        for connection in self.active_connections:
            await connection.send_json(message)

      .then(res => res.json())
      .then(data => setAlerts(data))

    const ws = new WebSocket(WS_URL)

    ws.onopen = () => setStatus("CONNECTED")
    ws.onclose = () => setStatus("DISCONNECTED")

    type: str

@app.post("/webrtc/offer")
async def start_session(
    offer: RTCOffer, 
    session_id: str = Body(..., embed=True),
    token: str = Body(..., embed=True)
):
    """
    1. Receive SDP Offer from Cockpit.
    2. Start FFMPEG Screen Capture.
    3. Return SDP Answer.
    """

    # if not verify_token(token): raise HTTPException(401)

        # Create the Peer Connection & Start Stream
        # We pass 'handle_input_event' so the DataChannel knows how to move the mouse
        result = await handle_offer(

            offer_type=offer.type, 
            session_id=session_id,
